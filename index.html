Question. What is dynamic programming?

Dynamic Programming is mainly an optimization over plain recursion. Wherever we see a recursive solution that has repeated calls for same inputs, we can optimize it using Dynamic Programming. The idea is to simply store the results of subproblems, so that we do not have to re-compute them when needed later. This simple optimization reduces time complexities from exponential to polynomial.

Types of dynamic programming problem
  1. 0/1 Knapsack problems
  2. Unbounded knapsack problems
  3. Longest common subsequence problems
  4. Matrix chain multiplication problems
0/1 knapsack problem
Subset sum problem
                        
                            int subSetProblem(int[] array, int length, int totalSum){
                                if (totalSum == 0){
                                    return 1;
                                }
                                else if (totalSum != 0 && length == 0){
                                    return 0;
                                }

                                if (dp[length][totalSum] != 0){
                                    return dp[length][totalSum];
                                }

                                if (totalSum > array[length-1])
                                    return dp[length][totalSum] = subSetProblem(array, length-1, totalSum-array[length-1]) |
                                                                    subSetProblem(array, length-1, totalSum);
                                else 
                                return dp[length][totalSum] = subSetProblem(array, length-1, totalSum);
                            }
Maximize profit problem
                        
                            int knapsackProblem(int[] value, int[] wt, int weight, int length){
                                if (length == 0 && weight == 0){
                                    return 0;
                                }
                                if (dp[length-1][weight-1] != -1){
                                    return dp[length-1][weight-1];
                                }
                                if (weight > wt[length-1]){
                                    return dp[length-1][weight-1] = Math.max(value[length-1]+knapsackProblem(value,wt, weight-wt[length-1],length-1), knapsackProblem(value, wt, weight, length-1));
                                }
                                else {
                                    return dp[length-1][weight-1] = knapsackProblem(value, wt, weight, length-1);
                                }
                            }
Unbounded knapsack problem
Coin change problem
                        
                            int coinChangeProblem(int[] value, int length, int totalValue){
                                if (totalValue == 0){
                                    return 1;
                                }
                                else if (totalValue != 0 && length == 0){
                                    return 0;
                                }

                                if (dp[length][totalValue] != 0){
                                    return dp[length][totalValue];
                                }

                                if (totalValue > value[length-1])
                                    return dp[length][totalValue] = coinChangeProblem(value, length, totalValue - value[length-1]) + coinChangeProblem(value, length-1, totalValue);
                                else 
                                    return dp[length][totalValue] = coinChangeProblem(value, length-1, totalValue);
                            }
Rode cutting problem
                        
                            int knapsackProblem(int[] price, int[] length, int totalLength, int lengthOfArray){
                                if (lengthOfArray == 0 && totalLength == 0){
                                    return 0;
                                }
                                if (dp[lengthOfArray][totalLength] != -1){
                                    return dp[lengthOfArray -1][totalLength -1];
                                }
                                if (totalLength > length[lengthOfArray -1]){
                                    return dp[lengthOfArray][totalLength] = Math.max(price[lengthOfArray -1]+knapsackProblem(price, length, totalLength - length[lengthOfArray], lengthOfArray), knapsackProblem(price, length, totalLength, lengthOfArray -1));
                                }
                                else {
                                    return dp[lengthOfArray][totalLength] = knapsackProblem(price, length, totalLength, lengthOfArray -1);
                                }
                            }
Longest common subsequence problems
Longest common subsequence problem
                        
                            int findLongestCommonSubsequenceUsingTopDown(char[] str1, int m, char[] str2, int n){
                                int[] dp = new int[m][n];

                                for (int i = 0; i < m; i++) {
                                    dp[i][0] = 0;
                                }
                                for (int i = 0; i < m; i++) {
                                    for (int j = 0; j < n; j++) {
                                        if (str1[i-1] == str2[j-1])
                                            dp[i][j] = 1+dp[i-1][j-1];
                                        else 
                                            dp[i][j] = Math.max(dp[i][j-1],dp[i-1][j]);
                                    }
                                }

                                return dp[m][n];
                            }
Longest common substring problem
                        
                            int findLongestCommonString(char[] str1, int m, char[] str2, int n){
                                if (n==0 && m==0){
                                    return 0;
                                }
                                if (dp[m][n] == -1)
                                    return dp[m][n];
                                if (str1[m-1] == str2[n-1])
                                    return dp[m][n] = 1+findLongestCommonString(str1,m-1, str2, n-1);
                                else 
                                    return dp[m][n] = 0;
                            }
Longest palindromic subsequence problem
                        
                            int longestPalindromicSubsequence(String str){
                                char[] charArray = str.toCharArray();
                                char[] reverseCharArray = new char[str.length()];
                                int k=0;
                                for (int i = str.length()-1; i >=0 ; i--) {
                                    reverseCharArray[k++]  = charArray[i];
                                }
                                return findLongestCommonSubsequence(charArray,str.length(),reverseCharArray,str.length());
                            }
Shortest common super sequence problem
                        
                            int shortestCommonSuperSequence(String str1, String str2){
                                int length = findLongestCommonSubsequence(str1.toCharArray(),str1.length(),str2.toCharArray(),str2.length());
                                return str1.length()+str2.length()-length;
                            }
Matrix chain multiplication problems
Matrix chain multiplication problem
                        
                            int findMatrixChainMultiplication(int[] array, int left, int right){
                               if (left <= right){
                                   return 0;
                               }
                                if (dp[left][right] != -1)
                                    return dp[left][right];
                               int minimumMultiplication =Integer.MAX_VALUE;

                               for (int i = left; i int temp = findMatrixChainMultiplication(array,left,i)+findMatrixChainMultiplication(array,i+1,right)+array[left]*array[i]*array[right];
                                                           if (temp < minimumMultiplication)
                                       minimumMultiplication = temp;
                               }

                               return dp[left][right] = minimumMultiplication;
                           }
Palindrome partitioning problem
                        
                            int findPartitionToMakePalindrome(char[] charArray, int left, int right){
                                if (left >= right)
                                    return 0;
                                if (findPalindrome(charArray))
                                    return 0;
                                if (dp[left][right] != -1)
                                    return dp[left][right];
                                int minimumPartition = Integer.MAX_VALUE;
                                for (int i = left; i < right; i++) {
                                    int temp = findPartitionToMakePalindrome(charArray,left,i)+findPartitionToMakePalindrome(charArray,i+1, right)+1;
                                    if (temp < minimumPartition)
                                        minimumPartition = temp;
                                }
                                return dp[left][right] = minimumPartition;
                            }
Evaluate boolean expression problem
                        
                            int findNumberOfPartitionToMakeExpressionTrue(char[] charArray, int left, int right, char isTrue){
                                if (left > right)
                                    return 0;
                                if (left == right){
                                    if (isTrue == charArray[left])
                                        return 1;
                                    else 
                                        return 0;
                                }
                                String key = left+" "+right+" "+isTrue;
                                if (dp.containsKey(key))
                                    return dp.get(key);
                                int ans=0;
                                for (int i = left+1; i < left; i+=2) {
                                    int leftTrue = findNumberOfPartitionToMakeExpressionTrue(charArray, left, i-1, 'T');
                                    int leftFalse = findNumberOfPartitionToMakeExpressionTrue(charArray, left, i-1, 'F');
                                    int rightTrue = findNumberOfPartitionToMakeExpressionTrue(charArray, i+1, right, 'T');
                                    int rightFalse = findNumberOfPartitionToMakeExpressionTrue(charArray, i+1, right, 'F');
                                    switch (charArray[i]){
                                        case '&':
                                            if (isTrue == 'T')
                                                ans += leftTrue & rightTrue;
                                            else 
                                                ans += leftTrue & rightFalse + leftFalse & rightTrue + leftFalse & rightFalse;
                                            break;
                                        case '|':
                                            if (isTrue == 'T')
                                                ans += leftTrue | rightTrue + leftFalse | rightTrue + leftTrue | rightFalse;
                                            else 
                                                ans += leftFalse | rightFalse;
                                            break;
                                        case '^':
                                            if (isTrue == 'T')
                                                ans += leftTrue ^ rightFalse + leftFalse ^ rightTrue;
                                            else 
                                                ans += leftTrue ^ rightTrue + leftFalse ^ rightFalse;
                                            break;
                                    }
                                }
                                dp.put(key,ans);
                                return ans;
                            }
